
//If a hopper extra filled,
//you can empty it with reject system.

IF M66 = 1 THEN
	
	FOR fazla:= 0 TO 11 BY 1 DO
	RST(1,M44);
	RST(1,M55);
	OUT(LC_DIZI[fazla] >= D100*2 ,KEFE_DIZI[fazla]);
	
	//500ms open, for reject
	TON_1(IN:=KEFE_DIZI[fazla],PT:= T#500ms);
	
	//After reject, restart working
	OUT(1,M44);
	OUT(1,M55);
	RST(1,M66);
	
	END_FOR;
	
END_IF;
