//FAZLA DOLAN KEFEYI ASAGI BOSALTMAK ICIN
//KEFEMOV PROGRAMINDA 100 SNDE 1 DEFA SETLENIR
IF REJECT = 1 THEN
	RST(1,HESAPLA);//hesaplama ve dolumu durdur, reject yapilacak.
	RST(1,DOLUMACIK);
	
	FOR fazla:= 0 TO 11 BY 1 DO //0dan11e kadar kontrol et
	
		RST( LC_DIZI[fazla] > (D300*3/2),KEFE_DIZI[fazla]); // burada fazla dolu olmayan kefeleri tespit edip, aciksa kapatir.	
		KEFE_DIZI[fazla] := REJECT AND LC_DIZI[fazla] >= D300*3/2; //REJECT BITI 1 ISE FAZLA DOLAN KEFEYI AC
		
		HESAPLA := 0; //REJECT YAPARKEN KEFE HESAPLAMAYI DEVRE DISI BIRAK
		MOV(LC_DIZI[fazla] >= D300*3/2, 0, LC_DIZI[fazla]); //SIMULASYON ICIN KEFE AGRLK SIFIRLA
		
	
	//REJECT YAPILDIKTAN SONRA, TEKRAR NORMALE DONUS
	OUT(0,REJECTTETIK);
	OUT(1,DOLUMACIK);
	RST(1,REJECT);
	IF HESAPLA = 1 THEN
		EXIT;
	END_IF;
	
	END_FOR;
END_IF;
